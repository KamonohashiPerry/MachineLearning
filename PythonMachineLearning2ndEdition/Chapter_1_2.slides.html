    def fit(self, X, y):
        rgen = np.random.RandomState(self.random_state) # 
        self.w_ = rgen.normal(loc=0.0, # 
                                               scale=0.01, # 
                                               size=1 + X.shape[1] # 1X.shape[1] 
                                             )
        self.errors_ = [] # 
        
        for _ in range(self.n_iter): # _
            errors = 0
            for xi, target in zip(X, y): # zip
                update = self.eta * (target - self.predict(xi)) # 
                self.w_[1:] += update * xi # 
                self.w_[0] += update # xi1
                errors += int(update != 0.0)  # 0
            self.errors_.append(errors) # append
        return self # 
    
    def net_input(self, X):
        return np.dot(X, self.w_[1:]) + self.w_[0] # 
    
    def predict(self, X):
        return np.where(self.net_input(X) >= 0.0, 1, -1) # 
''';

勾配降下法¶


ADALINE $$ J(W) = \frac{1}{2} \sum _i (y^{(i)} - \phi (z^{(i)}))^2 $$
$$ w := w + \Delta w \\ \Delta w = - \eta \nabla J (w) $$ J(w)
$$ \frac{\partial J}{\partial w_j} = - \sum_i (y^{(i)} - \phi (z^{(i)}) )x^{(i)}_j $$
$$ \Delta w_j = - \eta \frac{\partial J}{\partial w_j} = \eta \sum_i (y^{(i)} - \phi (z^{(i)}) )x^{(i)}_j $$ summation

    def fit(self, X, y):
        rgen = np.random.RandomState(self.random_state)
        self.w_ = rgen.normal(loc=0.0,
                                               scale=0.01,
                                               size=1+X.shape[1]
                                               )
        self.cost_ = []
        
        for i in range(self.n_iter):
            net_input = self.net_input(X)
            output = self.activation(net_input) # 
            errors = (y - output) # 
            self.w_[1:] += self.eta * X.T.dot(errors) # 
            self.w_[0] += self.eta * errors.sum() # 
            cost = (errors**2).sum() / 2.0 # 2
            self.cost_.append(cost)
        return self
    
    def net_input(self, X):
        return np.dot(X, self.w_[1:]) + self.w_[0] # 
    
    def activation(self, X): 
        return X # 
    
    def predict(self, X):
        return np.where(self.activation(self.net_input(X)) >= 0.0, 1, -1) # 
''';

from numpy.random import seed

class AdalineSGD(object):
    def __init__(self, eta=0.01, n_iter=10, shuffle=True, random_state=None):
        self.eta = eta
        self.n_iter = n_iter
        self.w_initialized = False
        self.shuffle = shuffle
        self.random_state = random_state
        
    def fit(self, X, y):
        self._initialize_weights(X.shape[1]) # 
        self.cost_ = [] # 1
        for i in range(self.n_iter):
            if self.shuffle: # shuffleBoolean
                X, y = self._shuffle(X, y) # 
            cost = [] # 2
            for xi, target in zip(X, y):
                cost.append(self._update_weights(xi,target)) # 
            avg_cost = sum(cost)/len(y) # 
            self.cost_.append(avg_cost) # append
        return self
''';

    def partial_fit(self, X, y):
        if not self.w_initialized:
            self._initialize_weights(X.shape[1]) # 
        if y.ravel().shape[0] > 1: # y1
            for xi, target  in zip(X, y):
                self._update_weights(xi,target) # 
        else:
            self._update_weights(X, y) # y11
        return self
    
    def _shuffle(self, X, y):
        r = self.rgen.permutation(len(y)) # 
        return X[r], y[r]
    
    def _initialize_weights(self, m): # 
        self.rgen = np.random.RandomState(self.random_state)
        self.w_ = self.rgen.normal(loc=0.0, scale=0.01, size=1 + m)
        self.w_initialized = True
